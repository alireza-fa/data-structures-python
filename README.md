class Node:
    def __init__(self, data):
        self.data = data
        self.next = None


class SinglyLinkedList:

    def __init__(self) -> None:
        self.__head = None
        self.__tail = None
        self.__size = 0

    def __set_new_head(self, node: Node) -> None:
        self.__head = node

    def __set_new_tail(self, node: Node) -> None:
        self.__tail = node

    def head(self) -> Node:
        return self.__head

        return self.__tail

        if self.head() is None:
